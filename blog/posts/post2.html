class GrafoMatriz:
    def __init__(self, n):
        self.n = n
        self.matriz = [[0] * n for _ in range(n)]
    
    def agregar_arista(self, u, v, peso=1):
        """Agrega una arista entre u y v"""
        self.matriz[u][v] = peso
        self.matriz[v][u] = peso  # Para grafos no dirigidos
    
    def existe_arista(self, u, v):
        """Verifica si existe arista entre u y v"""
        return self.matriz[u][v] != 0
    
    def obtener_vecinos(self, u):
        """Retorna todos los vecinos del v√©rtice u"""

class GrafoLista:
    def __init__(self, n):
        self.n = n
        self.lista = [[] for _ in range(n)]
    
    def agregar_arista(self, u, v, peso=1):
        """Agrega una arista entre u y v"""
        self.lista[u].append((v, peso))
        self.lista[v].append((u, peso))  # Para grafos no dirigidos
    
    def existe_arista(self, u, v):
        """Verifica si existe arista entre u y v"""
        for vecino, _ in self.lista[u]:
            if vecino == v:
                return True
        return False
    
    def obtener_vecinos(self, u):
        """Retorna todos los vecinos del v√©rtice u"""
        return [vecino for vecino, _ in self.lista[u]]

# Uso
grafo = GrafoLista(5)
grafo.agregar_arista(0, 1)  # Arista entre A(0) y B(1)
grafo.agregar_arista(1, 2)  # Arista entre B(1) y C(2)
print(grafo.existe_arista(0, 1))  # True
